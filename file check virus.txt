wireshark

# Monitor All Traffic
ip.addr != 192.168.2.1 && !dns && !ntp

# Suspicious Traffic
tcp.flags.reset == 1

# forced connection resets (someone cutting your connection)
tcp.flags.syn == 1 && tcp.flags.ack == 0

#  port scanning (hacker mapping your system)
tcp.analysis.retransmission

# Data Theft Detection
#large data leaving your network to outside IP
tcp.len > 1000 && ip.dst != 192.168.2.1

# data being pushed to external servers
!(ip.dst == 192.168.0.0/16) && tcp.flags.push == 1

#FTP transfers
ftp || ftp-data

# Unauthorized Access Detection  Metasploit backdoor port
tcp.port == 4444 or 3389 or 5900 or 22

#DNS Attack Detection
dns

# DNS errors — possible DNS poisoning
dns.flags.rcode != 0

# DNS going to unknown servers = very suspicious
dns && ip.dst != 8.8.8.8 && ip.dst != 1.1.1.1

#ARP Spoofing (Man in Middle) Detection
#two different MAC addresses
arp

# ARP spoof alert
arp.duplicate-address-detected

# ARP poisoning attack
arp.opcode == 2

# Filter for idle suspicious traffic
!(arp) && !(dns) && !(mdns) && !(ssdp) && !(nbns)
